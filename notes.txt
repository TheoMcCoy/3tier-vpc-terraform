resource "aws_db_instance" "database" {
  identifier        = "${local.project}-db-instance"
  engine            = "postgres"
  engine_version    = "15.17"
  instance_class    = "db.t3.micro" # Free Tier eligible instance class
  allocated_storage = 20            # Free Tier allows up to 20 GB (SSD)

  db_name  = "appdb"
  username = "dbadmin"
  password = var.db_master_password

  db_subnet_group_name   = aws_db_subnet_group.db.name
  vpc_security_group_ids = [aws_security_group.db.id]

  # Critical Free Tier settings to avoid the error you saw
  backup_retention_period = 1    # Must be 1 to stay within restricted limits
  skip_final_snapshot     = true # Prevents costs for storage after deletion
  multi_az                = false # Free Tier only covers Single-AZ instances
  publicly_accessible     = false

  tags = local.common_tags
}
